if len(pas)< 8:
    print("Пароль слишком короткий")
if pas.upper() == pas:
    print("Не содержит заглавных букв")
if pas.islower() == pas:
    print("Не содержит строчных букв")
if not any(char.isdigit() for char in pas):
    print("Не содержит цифр")
pac = "!@#№$%^&?*"
if not any(char in pac for char in pas):
    print("Не содержит спец символы")
if (len(pas) >= 8 and pas.upper != pas and pas.islower != pas and any(char.isdigit()for char in pas) and any(char in pac for char in pas)):
    print("Пароль надёжен")
